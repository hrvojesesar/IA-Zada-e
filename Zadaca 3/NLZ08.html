globals [ pomocna pomocna2 ]

breed [ arrow arrows ]
breed [ arrowkid arrowkids ]
breed [ circle circles ]

to setup
  clear-all
  reset-ticks
  create-arrow 1[
    set shape "arrow"
    set size 1
    set color red
    setxy -8 -6
    set heading 0

  ]

  create-arrowkid 1[
    set shape "arrow"
    set size 0.5
    set color yellow
    setxy -8 -6
    set heading 0

  ]
  create-arrow 1[
    set shape "arrow"
    set size 1
    set color red
    setxy 8 -6
    set heading 0

  ]

  create-arrowkid 1[
    set shape "arrow"
    set size 0.5
    set color yellow
    setxy 8 6
    set heading 0

  ]

  create-circle 2[
  set shape "circle"
    set color blue
    set size 1
    setxy random-pxcor random-pycor
  ]
end

to go
  if not any? turtles with [breed = circle][
    stop
  ]
  parent-arrow
  parent1-arrow
  ask circle[
    circle-walk
  ]

  tick
end

to parent-arrow
   if any? turtles with [breed = circle][
    ask turtle 0[
      set pomocna min-one-of patches with [any? turtles-here with [breed = circle]][distance myself]
      if pomocna != nobody[
        face pomocna
        fd 1
        if any? circle-here [
           kid
        ]
      ]
    ]
  ]
end

to kid
  ask turtle 1[
    while [patch-here != pomocna][
      face pomocna
      fd 1
    ]
    if patch-here = pomocna[
      ask circle-here[
            die
          ]
      while[patch-here != patch -8 -6][
        face patch -8 -6
        fd 1

      ]
    ]
  ]
end

to parent1-arrow
   if any? turtles with [breed = circle][
    ask turtle 2[
      set pomocna2 min-one-of patches with [any? turtles-here with [breed = circle]][distance myself]
      if pomocna2 != nobody[
        face pomocna2
        fd 1
        if any? circle-here [
           kid2
        ]
      ]
    ]
  ]
end

to kid2
  ask turtle 3[
    while [patch-here != pomocna2][
      face pomocna2
      fd 1
    ]
    if patch-here = pomocna2[
      ask circle-here[
            die
          ]
      while[patch-here != patch 8 6][
        face patch 8 6
        fd 1
      ]
    ]
  ]
end

to circle-walk
  set heading random 360
  fd 1
end
